## Complete Mapping Example

```typescript
import {
  floydSpinnerMapping,
  customFloydSpinners,
  cliSpinners,
} from './floyd-spinners.js';

// Get all spinner configs
const allSpinners = {
  ...cliSpinners,
  ...customFloydSpinners,
};

// Get spinner for any message
function getSpinner(message: string) {
  const spinnerKey = floydSpinnerMapping[message];
  return spinnerKey ? allSpinners[spinnerKey] : allSpinners.dots;
}
```
